//PRIMERA TABLA
CREATE TABLE gym_moderno.PRUEBA_CLIENTES(
NUM_IDENTIFICACION int,
NOMBRES VARCHAR(50),
APELLIDOS VARCHAR(50),
EMAIL VARCHAR(100),
CELULAR VARCHAR(10),
ALTURA int,
FECHA_INSCRIPCION DATE,

PRIMARY KEY (NUM_IDENTIFICACION)

);

SELECT * FROM gym_moderno.prueba_clientes;

//SEGUNDA TABLA
CREATE TABLE gym_moderno.PRUEBA_IMC(
ID INT AUTO_INCREMENT,
FECHA_INSCRIPCION DATE,
PESO_ACTUAL int,
DIAGNOSTICO VARCHAR(100),
NUM_IDENTIFICACION int,

PRIMARY KEY (ID)

);

SELECT * FROM gym_moderno.prueba_imc;

ALTER TABLE gym_moderno.prueba_imc ADD FOREIGN KEY (NUM_IDENTIFICACION) REFERENCES gym_moderno.prueba_clientes (NUM_IDENTIFICACION) ;
